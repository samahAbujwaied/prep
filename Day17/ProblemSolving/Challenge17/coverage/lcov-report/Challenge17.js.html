const isPalindrome = (str) => {
    // write your code here
    let a = /[^A-Za-z0-9]/g;
    str = str.toLowerCase().replace(a, "");
    for (let i = 0; i < str.length / 2; i++) {
      if (str[i] !== str[str.length - 1 - i]) {
        return false;
      }
    }
    return true;
}
